import urllib.request
from urllib.parse import urlparse, urljoin
import re

def extract_links(url):
    response = urllib.request.urlopen(url)
    if response.getcode() == 200:
        page_content = response.read().decode('utf-8')
        links = re.findall(r'href="(.*?)"', page_content)
        return links
    else:
        print("Error accessing:", url)
        return []

def check_vulnerabilities(url):
    response = urllib.request.urlopen(url)
    if response.getcode() == 200:
        page_content = response.read().decode('utf-8')
        if re.search(r'jquery\.js\?ver=1\.12\.4', page_content):
            print("Vulnerability found: Outdated jQuery version detected on", url)
        if re.search(r'SQL error', page_content, re.IGNORECASE):
            print("Vulnerability found: Possible SQL Injection on", url)
    else:
        print("Error accessing:", url)

def vulnerability_scanner(start_url):
    visited_urls = set()
    queue = [start_url]

    while queue:
        url = queue.pop(0)
        visited_urls.add(url)

        print("Scanning:", url)
        links = extract_links(url)
        for link in links:
            absolute_link = urljoin(url, link)
            if absolute_link not in visited_urls and absolute_link.startswith(start_url):
                queue.append(absolute_link)
                check_vulnerabilities(absolute_link)

start_url = "https://example.com"
vulnerability_scanner(start_url)

Web Vulnerability Scanner

This Python script acts as a web vulnerability scanner, empowering users to assess their website's security by detecting potential vulnerabilities like outdated jQuery versions and SQL injection risks.

Usage:

The vulnerability_scanner function requires one argument:

  • start_url: The URL from which the scanning process will begin.

Steps:

  1. Initiate Scanning: Start by calling the vulnerability_scanner function with the URL of the webpage you want to scan.

Functionality:

The vulnerability_scanner function executes the following:

  • Web Crawling: Traverses the webpage starting from the specified URL, scanning all linked pages within the same domain.
  • Link Extraction: Extracts links from each webpage, ensuring a comprehensive examination of your website's structure.
  • Vulnerability Assessment: Inspects each webpage for potential vulnerabilities, such as outdated jQuery versions and SQL injection risks.
  • Informative Output: If any vulnerabilities are found, the scanner promptly notifies you, providing detailed information about the vulnerability type and the URL where it was detected.

Example Usage:

start_url = "https://example.com"
vulnerability_scanner(start_url)

Important Note:

Ensure that the start_url variable in the provided Python script is replaced with the URL of the webpage you intend to scan.
